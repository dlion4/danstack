4.4.1 — Official SDKs

Language-specific libraries for fast, type-safe API integration.

PayMo Node.js SDK

Fully typed with TypeScript, Promise-based, Browser & Node environments supported.

npm install @paymo/paymo-node
import { PayMoClient } from '@paymo/paymo-node'; const paymo = new PayMoClient({ apiKey: process.env.PAYMO_SECRET_KEY, environment: 'sandbox' // or 'production' }); // Initiate an M-Pesa STK Push const response = await paymo.collections.requestMpesaPayment({ amount: 1500, phoneNumber: '254712345678', reference: 'INV-2025-001', description: 'Payment for web services' }); console.log(response.transactionId);
PayMo Python SDK

Python 3.7+ support, synchronous and async variations included.

pip install paymo-python
from paymo import PayMoClient paymo = PayMoClient(api_key="sk_test_12345") # Initiate PesaLink disbursement transfer = paymo.disbursements.create( amount=50000, currency="KES", destination=paymo.Destination.bank_account( bank_code="068", # Equity Bank account_number="0123456789" ), purpose="Salary payment" ) print(transfer.status)

Install Node.js / Browser SDK

Install via package manager:

npm install @paymo/paymo-node
# or
yarn add @paymo/paymo-node

Initialize client:

import { PayMoClient } from '@paymo/paymo-node';
const paymo = new PayMoClient({ apiKey: 'your_secret_key' });

Include via unpkg CDN in your HTML:

<script src="https://unpkg.com/@paymo/paymo-js@3.2.0/dist/paymo.min.js"></script>

Initialize global object:

const paymo = PayMo('your_public_key');
paymo.elements.create('checkout').mount('#payment-form');

Install the React library:

npm install @paymo/paymo-react
import { PayMoProvider, CheckoutElement } from '@paymo/paymo-react';

function App() {
return (
<PayMoProvider publicKey="pk_test_123">
<CheckoutElement amount={1500} onSuccess={handleSuccess} />
</PayMoProvider>
);
}

Webhook Signature Migration

Verify incoming PayMo webhooks using the x-paymo-signature header to ensure payloads originate from us and haven't been tampered with.

const crypto = require('crypto');

app.post('/webhook', express.raw({type: 'application/json'}), (req, res) => {
const signature = req.headers['x-paymo-signature'];
const hash = crypto.createHmac('sha256', process.env.PAYMO_WH_SECRET)
.update(req.body)
.digest('hex');
if (hash !== signature) return res.status(401).send('Invalid signature');

const event = JSON.parse(req.body);
// Handle event...
});
$payload = file_get_contents('php://input');
$signature = $_SERVER['HTTP_X_PAYMO_SIGNATURE'] ?? '';
$hash = hash_hmac('sha256', $payload, getenv('PAYMO_WH_SECRET'));

if (!hash_equals($hash, $signature)) {
http_response_code(401);
exit('Invalid signature');
}
Drop-in Checkout Widget

Embed a secure payment form directly on your site without handling sensitive routing.

<script src="https://js.paymo.com/v1/widget.js"></script>
<button id="pay-btn">Pay KES 1500</button>

<script>
const widget = PayMoWidget({
publicKey: 'pk_test_12345',
amount: 1500,
currency: 'KES',
onSuccess: function(ref) { alert('Paid: ' + ref); }
});
document.getElementById('pay-btn').onclick = () => widget.open();
</script>
